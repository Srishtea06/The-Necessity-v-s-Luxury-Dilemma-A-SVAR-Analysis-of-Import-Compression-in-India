print("--- STEP 1: LOADING DATA ---")
filename = 'itf new data.csv'

try:
    # Load CSV, skipping the second row (units)
    df = pd.read_csv(r'E:\Personal\SHAKSHI\final year MSC\term paper data\itf new data.csv', header=0, skiprows=[1])

    rename_map = {
        'Time Period': 'date',

    df = df.set_index('date').sort_index()

    cols = ['lbir', 'awcmr', 'er', 'oilimp', 'noilimp']
    df = df[cols].apply(pd.to_numeric, errors='coerce').dropna()
    
    print(f"Success! Data Loaded. Time Range: {df.index.min().date()} to {df.index.max().date()}")
    print(f"Total Observations: {len(df)}")
    print(df.head())


    print(df.head())

except FileNotFoundError:
    print(f"ERROR: Could not find '{filename}'. Make sure the file is in the same folder as this notebook.")
    # Stop execution if file missing
    raise

print("\n--- STEP 2: TRANSFORMATIONS ---")
data = pd.DataFrame()

# Interest Rates -> First Difference
data['lbir_diff'] = df['lbir'].diff()
data['awcmr_diff'] = df['awcmr'].diff()

# Trade/Price -> Log Growth Rates
data['er_growth'] = np.log(df['er']).diff() * 100
data['oil_growth'] = np.log(df['oilimp']).diff() * 100
data['noil_growth'] = np.log(df['noilimp']).diff() * 100

data = data.dropna()
print("Data transformed to stationary growth rates.")

# Check Stationarity (ADF Test)
print("\n--- ADF Stationarity Test Results ---")
for col in data.columns:
    res = adfuller(data[col])
    print(f"{col:15} | p-value: {res[1]:.5f} | {'Stationary' if res[1]<0.05 else 'Non-Stationary'}")

# ==========================================
# 4. VAR MODEL ESTIMATION
# ==========================================
print("\n--- STEP 3: VAR MODEL ESTIMATION ---")

# RECURSIVE ORDERING (Cholesky)
# 1. Global Shock (lbir)
# 2. Policy (awcmr)
# 3. Transmission (er)
# 4. Inelastic (oil)
# 5. Elastic (noil)
ordering = ['lbir_diff', 'awcmr_diff', 'er_growth', 'oil_growth', 'noil_growth']
var_data = data[ordering]

model = VAR(var_data)
lag_selection = model.select_order(maxlags=8)
optimal_lag = lag_selection.aic
print(f"\nOptimal Lag Length (AIC): {optimal_lag}")

results = model.fit(optimal_lag)
print("\nModel Summary:")
# Summary can be long, printing just the correlation of residuals
print(results.resid.corr())

# Check Stability
is_stable = results.is_stable()
print(f"\nModel Stability: {'Stable' if is_stable else 'Unstable (Check Data)'}")

            # Extract p-value of F-test at Lag 1 (or min p-value across lags)
            p_values = [test_result[i+1][0]['ssr_ftest'][1] for i in range(4)]
            min_p_val = min(p_values)
            granger_matrix.loc[r, c] = min_p_val
        else:
            granger_matrix.loc[r, c] = np.nan
